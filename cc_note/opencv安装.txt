# download opencv-2.4.11
wget -c http://downloads.sourceforge.net/project/opencvlibrary/opencv-unix/2.4.13/opencv-2.4.13.zip

# install dependencies
cd ~/libs
unzip opencv-2.4.13.zip
cd opencv-2.4.13
mkdir release   //不能是其它名字，在这里被坑了很久，以后还是老老实实照办吧，等有足够知识了再diy
cd release
mkdir installed

# compile and install


cd release

cmake -G "Unix Makefiles" -DCMAKE_CXX_COMPILER=/usr/bin/g++ CMAKE_C_COMPILER=/usr/bin/gcc -DCMAKE_BUILD_TYPE=RELEASE -DCMAKE_INSTALL_PREFIX=/home/chencheng/libs/opencv-2.4.13/release/installed -DWITH_TBB=ON -DBUILD_NEW_PYTHON_SUPPORT=ON -DWITH_V4L=ON  -DBUILD_EXAMPLES=ON -DWITH_OPENGL=ON -DINSTALL_TO_MANGLED_PATHS=ON -DINSTALL_CREATE_DISTRIB=ON -DENABLE_FAST_MATH=ON -DWITH_IMAGEIO=ON -DWITH_GSTREAMER=ON ..

make all -j23 # 24 cores
sudo make install

#修改Makefile.config
INCLUDE_DIRS := $(PYTHON_INCLUDE) /home/chencheng/libs/opencv-2.4.13/release/installed/include /usr/local/include
LIBRARY_DIRS := $(PYTHON_LIB) /home/chencheng/libs/opencv-2.4.13/release/installed/lib /home/chencheng/libs/opencv-2.4.13/release/installed/bin /usr/local/lib /usr/lib

#在.bashrc里添加opencv路径
export LD_LIBRARY_PATH=/home/chencheng/libs/opencv-2.4.13/release/installed/lib:$LD_LIBRARY_PATH


#为了能够import到cv2
